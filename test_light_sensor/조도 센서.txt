#define RELAY_PIN 33 // 릴레이에 연결된 핀 번호
const int sensorPin = A0; // 조도 센서의 핀 번호

void setup() {
	pinMode(RELAY_PIN, OUTPUT); // 릴레이 핀을 출력으로 설정
	Serial.begin(9600); // 시리얼 통신 시작
}

void loop() {
	int sensorValue = analogRead(sensorPin); // 조도 센서의 값 읽기
	Serial.print("Light Sensor Value: "); // 시리얼 모니터에 출력
	Serial.println(sensorValue); // 시리얼 모니터에 조도 센서 값 출력
	delay(1000); // 1초 대기

	if(sensorValue > 300)
		digitalWrite(RELAY_PIN, HIGH); // 빛이 어두우면 등을 켜다
	else
		digitalWrite(RELAY_PIN, LOW); // 빛이 밝으면 등을 끄다
}
